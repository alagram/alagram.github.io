class BankAccount
  @@total_accounts = 0

  def initialize
    @@total_accounts += 1
  end

  def self.accounts_created
    @@total_accounts
  end
end

savings = BankAccount.new
current = BankAccount.new
inv = BankAccount.new

puts "Number of accounts created is #{BankAccount.accounts_created}." 
# => Number of accounts created is 3.
