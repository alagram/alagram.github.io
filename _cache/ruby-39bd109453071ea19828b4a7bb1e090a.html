class Expense < ActiveRecord::Base
  has_many :images
  accepts_nested_attributes_for :images, :reject_if => proc { |attr| attr['image'].blank? }

  validate :must_have_image

  private

  def must_have_image
    errors[:base] << "Please upload image" if images_empty?
  end

  def images_empty?
    images.empty?
  end
end
